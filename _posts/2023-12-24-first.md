#CODE 작성 (2가지 방식)

*<pre><code>{}</code></pre>이용

<pre>  
<code>
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int main() {

	char* arr = (char*)malloc(10);
	int i = 0, temp;
	scanf("%s", arr);
	while (1)
	{
		if (*(arr + i) == '\0')
			break;
		i++;
	}
	for (int k = 0; k < i-1;k++)
	{
		for (int l = 0; l < (i - k) - 1; l++)
		{
			if (arr[l] < arr[l + 1])
			{
				temp = arr[l];
				arr[l] = arr[l + 1];
				arr[l + 1] = temp;
			}
		}
	}
	printf("%s", arr);
	return 0;
}
